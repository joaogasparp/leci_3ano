Comandos (ordenados pela maneira que se devem usar):
antlr4-test		//echo "hello bye algo joao" | antlr4-test Hello greetings -gui/-tree
antlr4-visitor		//antlr4-visitor Hello execute2 execute -> cria ficheiro .java execute2 do tipo execute para a gramátiva Hello
antlr4-main		//antlr4-main Hello greetings -v Execute -> cria class main .java com o nome da gramática e a regra principal, bem como o visitor que queremos
antlr4-build    	//compila o programa
antlr4-run		//echo "hello bye algo joao" | antlr4-run
antlr4-clean


1º -> antlr4-build
2º -> echo "2 2 + 2 *" | antlr4-test -tree  (para ver a árvore)
ou
2º -> echo "2 2 + 2 *" | antlr4-test -gui   (para ver a árvore gráfica)
3º -> antlr4-visitor SuffixCalculator Interpreter Double
4º -> antlr4-main SuffixCalculator program -v Interpreter
5º -> antlr4-build
6º -> echo "2 2 + 2 *" | antlr4-run     (caso de linha a linha)
ou
6º -> antlr4-run < input.txt            (caso de ficheiro)
